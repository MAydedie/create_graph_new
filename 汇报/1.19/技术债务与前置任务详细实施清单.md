```python
"""
统一数据访问层 - 封装所有分析结果的访问接口
"""
from typing import Dict, List, Optional, Any
from pathlib import Path
import os
import json
import threading
from collections import defaultdict

class DataAccessor:
    """统一数据访问器"""
    
    def __init__(self):
        # 缓存锁
        self._cache_lock = threading.Lock()
        # 功能层级分析结果缓存
        self._function_hierarchy_cache: Dict[str, Dict] = {}
        # 主分析结果缓存
        self._main_analysis_cache: Dict[str, Dict] = {}
        # 项目分析报告缓存
        self._report_cache: Dict[str, Any] = {}
    
    def load_function_hierarchy(self, project_path: str) -> Optional[Dict]:
        """加载功能层级分析结果"""
        normalized_path = os.path.normpath(project_path)
        with self._cache_lock:
            return self._function_hierarchy_cache.get(normalized_path)
    
    def save_function_hierarchy(self, project_path: str, data: Dict):
        """保存功能层级分析结果"""
        normalized_path = os.path.normpath(project_path)
        with self._cache_lock:
            self._function_hierarchy_cache[normalized_path] = data
    
    def get_partition_analysis(self, project_path: str, partition_id: str) -> Optional[Dict]:
        """获取指定分区的分析结果"""
        hierarchy_data = self.load_function_hierarchy(project_path)
        if not hierarchy_data:
            return None
        partition_analyses = hierarchy_data.get('partition_analyses', {})
        return partition_analyses.get(partition_id)
    
    def get_all_partitions(self, project_path: str) -> List[Dict]:
        """获取所有分区的列表"""
        hierarchy_data = self.load_function_hierarchy(project_path)
        if not hierarchy_data:
            return []
        partition_analyses = hierarchy_data.get('partition_analyses', {})
        return list(partition_analyses.values())
    
    def get_path_analyses(self, project_path: str, partition_id: str) -> List[Dict]:
        """获取指定分区的所有路径分析"""
        partition_data = self.get_partition_analysis(project_path, partition_id)
        if not partition_data:
            return []
        return partition_data.get('path_analyses', [])
    
    def get_experience_paths(self, project_path: str, partition_id: Optional[str] = None) -> List[Dict]:
        """
        获取经验路径列表（用于匹配）
        
        Args:
            project_path: 项目路径
            partition_id: 分区ID（如果为None，返回所有分区的路径）
        
        Returns:
            经验路径列表，每条路径包含：
            - path_id: 路径ID
            - partition_id: 所在分区ID
            - function_chain: 函数调用链
            - io_summary: I/O摘要
            - semantic_label: 语义标签（如果有）
        """
        if partition_id:
            # 返回指定分区的路径
            path_analyses = self.get_path_analyses(project_path, partition_id)
            return self._convert_path_analyses_to_experience_paths(path_analyses, partition_id)
        else:
            # 返回所有分区的路径
            all_paths = []
            partitions = self.get_all_partitions(project_path)
            for partition in partitions:
                partition_id = partition.get('partition_id')
                path_analyses = self.get_path_analyses(project_path, partition_id)
                paths = self._convert_path_analyses_to_experience_paths(path_analyses, partition_id)
                all_paths.extend(paths)
            return all_paths
    
    def _convert_path_analyses_to_experience_paths(self, path_analyses: List[Dict], partition_id: str) -> List[Dict]:
        """将路径分析数据转换为经验路径格式"""
        experience_paths = []
        for idx, path_analysis in enumerate(path_analyses):
            path_id = f"{partition_id}_path_{idx}"
            leaf_node = path_analysis.get('leaf_node', '')
            path_nodes = path_analysis.get('path', [])
            
            # 提取I/O摘要
            io_graph = path_analysis.get('io_graph', {})
            io_summary = self._extract_io_summary(io_graph)
            
            experience_path = {
                'path_id': path_id,
                'partition_id': partition_id,
                'function_chain': path_nodes,
                'leaf_node': leaf_node,
                'io_summary': io_summary,
                'semantic_label': path_analysis.get('description', ''),  # 如果路径有描述
                'cfg_data': path_analysis.get('cfg_data'),
                'dfg_data': path_analysis.get('dfg_data')
            }
            experience_paths.append(experience_path)
        return experience_paths
    
    def _extract_io_summary(self, io_graph: Dict) -> Dict[str, Any]:
        """从I/O图中提取摘要信息"""
        if not io_graph:
            return {'input': [], 'output': []}
        
        inputs = []
        outputs = []
        
        # 遍历I/O图的节点，识别输入和输出
        nodes = io_graph.get('nodes', [])
        edges = io_graph.get('edges', [])
        
        # 简单的启发式：找到没有入边的节点（输入）和没有出边的节点（输出）
        node_has_incoming = set()
        node_has_outgoing = set()
        
        for edge in edges:
            source = edge.get('source')
            target = edge.get('target')
            if source:
                node_has_outgoing.add(source)
            if target:
                node_has_incoming.add(target)
        
        for node in nodes:
            node_id = node.get('id', '')
            if node_id not in node_has_incoming:
                inputs.append(node.get('label', node_id))
            if node_id not in node_has_outgoing:
                outputs.append(node.get('label', node_id))
        
        return {
            'input': inputs[:5],  # 限制前5个
            'output': outputs[:5]
        }
    
    def load_project_report(self, project_path: str) -> Optional[Any]:
        """加载项目分析报告（ProjectAnalysisReport）"""
        normalized_path = os.path.normpath(project_path)
        with self._cache_lock:
            return self._report_cache.get(normalized_path)
    
    def save_project_report(self, project_path: str, report: Any):
        """保存项目分析报告"""
        normalized_path = os.path.normpath(project_path)
        with self._cache_lock:
            self._report_cache[normalized_path] = report


# 全局单例
_global_data_accessor: Optional[DataAccessor] = None
_data_accessor_lock = threading.Lock()

def get_data_accessor() -> DataAccessor:
    """获取全局DataAccessor实例（单例模式）"""
    global _global_data_accessor
    if _global_data_accessor is None:
        with _data_accessor_lock:
            if _global_data_accessor is None:
                _global_data_accessor = DataAccessor()
    return _global_data_accessor
```

"""
统一LLM调用封装 - 提供统一的LLM接口，支持Prompt模板、重试、缓存等
"""
from typing import Optional, Dict, Any, List, Callable
import os
import json
import logging
import time
from functools import lru_cache
from dataclasses import dataclass
from enum import Enum

logger = logging.getLogger(__name__)


class LLMProvider(Enum):
    """LLM提供商枚举"""
    DEEPSEEK = "deepseek"
    OPENAI = "openai"


@dataclass
class LLMConfig:
    """LLM配置"""
    api_key: str
    base_url: str = "https://api.deepseek.com/v1"
    model: str = "deepseek-chat"
    temperature: float = 0.3
    max_tokens: int = 8000
    timeout: int = 60
    max_retries: int = 3
    retry_delay: float = 1.0


class LLMHelper:
    """统一LLM调用封装"""
    
    def __init__(self, config: LLMConfig):
        self.config = config
        self.llm_client = self._init_llm_client()
        self._token_count = 0  # Token统计
        self._call_count = 0   # 调用次数统计
    
    def _init_llm_client(self):
        """初始化LLM客户端"""
        try:
            from langchain_openai import ChatOpenAI
            return ChatOpenAI(
                api_key=self.config.api_key,
                base_url=self.config.base_url,
                model=self.config.model,
                temperature=self.config.temperature,
                max_tokens=self.config.max_tokens,
                timeout=self.config.timeout
            )
        except ImportError:
            try:
                from langchain_community.chat_models import ChatOpenAI
                return ChatOpenAI(
                    openai_api_key=self.config.api_key,
                    openai_api_base=self.config.base_url,
                    model_name=self.config.model,
                    temperature=self.config.temperature,
                    max_tokens=self.config.max_tokens,
                    timeout=self.config.timeout
                )
            except ImportError:
                logger.warning("未安装langchain，将使用直接API调用模式")
                return None
    
    def call(
        self,
        system_prompt: str,
        user_prompt: str,
        cache_key: Optional[str] = None,
        use_cache: bool = True
    ) -> str:
        """
        调用LLM
        
        Args:
            system_prompt: 系统提示词
            user_prompt: 用户提示词
            cache_key: 缓存键（如果提供，会使用缓存）
            use_cache: 是否使用缓存
        
        Returns:
            LLM响应文本
        """
        # 检查缓存
        if use_cache and cache_key:
            cached_result = self._get_cache(cache_key)
            if cached_result:
                logger.debug(f"使用缓存结果: {cache_key}")
                return cached_result
        
        # 调用LLM（带重试）
        response = self._call_with_retry(system_prompt, user_prompt)
        
        # 保存缓存
        if use_cache and cache_key:
            self._save_cache(cache_key, response)
        
        # 更新统计
        self._call_count += 1
        # TODO: 估算Token数量（需要实际API返回才能准确统计）
        
        return response
    
    def _call_with_retry(self, system_prompt: str, user_prompt: str) -> str:
        """带重试机制的LLM调用"""
        last_error = None
        for attempt in range(self.config.max_retries):
            try:
                if self.llm_client:
                    from langchain_core.messages import SystemMessage, HumanMessage
                    messages = [
                        SystemMessage(content=system_prompt),
                        HumanMessage(content=user_prompt)
                    ]
                    response = self.llm_client.invoke(messages)
                    return response.content
                else:
                    # 直接API调用（fallback）
                    return self._call_api_directly(system_prompt, user_prompt)
            except Exception as e:
                last_error = e
                if attempt < self.config.max_retries - 1:
                    wait_time = self.config.retry_delay * (2 ** attempt)  # 指数退避
                    logger.warning(f"LLM调用失败（尝试 {attempt + 1}/{self.config.max_retries}），{wait_time}秒后重试: {e}")
                    time.sleep(wait_time)
                else:
                    logger.error(f"LLM调用最终失败: {e}")
                    raise
        
        raise Exception(f"LLM调用失败: {last_error}")
    
    def _call_api_directly(self, system_prompt: str, user_prompt: str) -> str:
        """直接API调用（fallback方案）"""
        import requests
        url = f"{self.config.base_url}/chat/completions"
        headers = {
            "Authorization": f"Bearer {self.config.api_key}",
            "Content-Type": "application/json"
        }
        data = {
            "model": self.config.model,
            "messages": [
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt}
            ],
            "temperature": self.config.temperature,
            "max_tokens": self.config.max_tokens
        }
        response = requests.post(url, json=data, headers=headers, timeout=self.config.timeout)
        response.raise_for_status()
        result = response.json()
        return result['choices'][0]['message']['content']
    
    def _get_cache(self, cache_key: str) -> Optional[str]:
        """获取缓存"""
        # TODO: 实现文件缓存或内存缓存
        return None
    
    def _save_cache(self, cache_key: str, value: str):
        """保存缓存"""
        # TODO: 实现文件缓存或内存缓存
        pass
    
    def get_statistics(self) -> Dict[str, Any]:
        """获取统计信息"""
        return {
            'call_count': self._call_count,
            'token_count': self._token_count,
            'average_tokens_per_call': self._token_count / self._call_count if self._call_count > 0 else 0
        }

1. **策略限制：** 当前从"叶子节点回溯到入口点"，可能遗漏：
   - 中间路径（既不从入口开始，也不在叶子结束）
   - 跨分区路径（功能涉及多个分区）
   - 长路径（被 `max_path_length=10` 截断）

2. **验证建议：**
   - 先用现有方法做验证测试（如测试"OAuth登录"是否能找到相关路径）
   - 如果覆盖率不足，再逐步改进

```python
"""
经验路径存储服务 - 负责经验路径的持久化和加载
"""
from typing import Dict, List, Optional, Any
from pathlib import Path
import json
import os
from datetime import datetime

class ExperiencePathStorage:
    """经验路径存储服务"""
    
    def __init__(self, storage_dir: str = "output_analysis/experience_paths"):
        """
        初始化存储服务
        
        Args:
            storage_dir: 存储目录
        """
        self.storage_dir = Path(storage_dir)
        self.storage_dir.mkdir(parents=True, exist_ok=True)
    
    def save_experience_paths(
        self,
        project_path: str,
        experience_paths: List[Dict],
        partition_analyses: Dict[str, Dict]
    ):
        """
        保存经验路径到JSON文件
        
        Args:
            project_path: 项目路径
            experience_paths: 经验路径列表（从DataAccessor获取）
            partition_analyses: 分区分析结果
        """
        # 按项目路径生成文件名（使用路径的hash避免特殊字符）
        import hashlib
        project_hash = hashlib.md5(project_path.encode()).hexdigest()[:8]
        project_name = os.path.basename(project_path) or "unknown_project"
        filename = f"{project_name}_{project_hash}.json"
        filepath = self.storage_dir / filename
        
        # 组织数据
        data = {
            'version': '0.1',
            'project_path': project_path,
            'project_name': project_name,
            'analysis_timestamp': datetime.now().isoformat(),
            'total_paths': len(experience_paths),
            'partitions': []
        }
        
        # 按分区组织路径
        partition_paths_map = {}
        for path in experience_paths:
            partition_id = path['partition_id']
            if partition_id not in partition_paths_map:
                partition_paths_map[partition_id] = []
            partition_paths_map[partition_id].append(path)
        
        # 构建分区数据
        for partition_id, paths in partition_paths_map.items():
            partition_data = partition_analyses.get(partition_id, {})
            partition_entry = {
                'partition_id': partition_id,
                'partition_name': partition_data.get('partition_name', partition_id),
                'total_paths': len(paths),
                'paths': paths
            }
            data['partitions'].append(partition_entry)
        
        # 保存到JSON
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False)
        
        print(f"[ExperiencePathStorage] ✅ 经验路径已保存到: {filepath}")
    
    def load_experience_paths(self, project_path: str) -> Optional[Dict]:
        """
        加载经验路径
        
        Args:
            project_path: 项目路径
        
        Returns:
            经验路径数据，如果不存在则返回None
        """
        # 查找匹配的JSON文件
        project_name = os.path.basename(project_path) or "unknown_project"
        
        for filepath in self.storage_dir.glob(f"{project_name}_*.json"):
            try:
                with open(filepath, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                
                # 验证项目路径是否匹配
                if data.get('project_path') == project_path or data.get('project_name') == project_name:
                    return data
            except Exception as e:
                print(f"[ExperiencePathStorage] ⚠️ 加载文件失败 {filepath}: {e}")
                continue
        
        return None
    
    def list_all_projects(self) -> List[Dict[str, str]]:
        """列出所有已存储的项目"""
        projects = []
        for filepath in self.storage_dir.glob("*.json"):
            try:
                with open(filepath, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    projects.append({
                        'project_path': data.get('project_path', ''),
                        'project_name': data.get('project_name', ''),
                        'analysis_timestamp': data.get('analysis_timestamp', ''),
                        'total_paths': data.get('total_paths', 0)
                    })
            except Exception as e:
                print(f"[ExperiencePathStorage] ⚠️ 读取文件失败 {filepath}: {e}")
        return projects
```
